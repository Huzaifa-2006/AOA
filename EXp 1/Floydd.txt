#include <stdio.h>
#define MAX 100
#define INF 999

int adj[MAX][MAX];
int n;

void floydWarshall() {
    int i, j, k;
    
    // Floyd-Warshall algorithm
    for(k = 0; k < n; k++) {
        for(i = 0; i < n; i++) {
            for(j = 0; j < n; j++) {
                if(adj[i][k] + adj[k][j] < adj[i][j]) {
                    adj[i][j] = adj[i][k] + adj[k][j];
                }
            }
        }
    }
}

void printSolution() {
    int i, j;
    printf("\nShortest distances between all pairs of vertices:\n");
    printf("     ");
    for(i = 0; i < n; i++) {
        printf("%4d ", i);
    }
    printf("\n");
    
    for(i = 0; i < n; i++) {
        printf("%4d ", i);
        for(j = 0; j < n; j++) {
            if(adj[i][j] == INF) {
                printf(" INF ");
            } else {
                printf("%4d ", adj[i][j]);
            }
        }
        printf("\n");
    }
}

int main() {
    int i, j;

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    printf("Enter adjacency matrix of the graph:\n");
    printf("(Use %d for no direct path between vertices)\n", INF);
    for(i = 0; i < n; i++) {
        for(j = 0; j < n; j++) {
            scanf("%d", &adj[i][j]);
        }
    }

    printf("\nOriginal adjacency matrix:\n");
    printSolution();

    floydWarshall();

    printf("\nAfter applying Floyd-Warshall algorithm:\n");
    printSolution();

    return 0;
}
